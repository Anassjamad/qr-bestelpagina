import mongoose from 'mongoose';

const MONGODB_URI = process.env.MONGODB_URI; // Veilig ophalen uit env variabelen

const connectDB = async () => {
  if (mongoose.connection.readyState === 0) {
    await mongoose.connect(MONGODB_URI);
  }
};

export default function handler(req, res) {
  if (req.method === 'POST') {
    // Je kunt hier de bestelling verwerken
    const { item, quantity } = req.body;

    // Voorbeeld van een respons
    res.status(200).json({ message: `Bestelling van ${quantity} ${item} ontvangen.` });
  } else {
    res.status(405).json({ error: 'Method Not Allowed' });
  }
}
